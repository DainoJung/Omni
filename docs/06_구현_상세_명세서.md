```python
# prompts/text_generation.py

from typing import Optional

PROMPT_MAP = {
    "food": FOOD_PROMPT,
    "fashion": FASHION_PROMPT,
}

def build_prompt(
    template_category: str,
    brand_name: str,
    description: str,
    event_period: Optional[str] = None,
    price_info: Optional[str] = None,
) -> str:
    """카테고리에 맞는 프롬프트를 선택하고 변수를 채운다."""

    template = PROMPT_MAP.get(template_category)
    if not template:
        raise ValueError(f"지원하지 않는 카테고리: {template_category}")

    return template.format(
        brand_name=brand_name,
        description=description,
        event_period=event_period or "미정",
        price_info=price_info or "미정",
    )
```

### 4.5 Gemini 응답 검증

```python
# services/gemini_service.py - 응답 파싱 및 검증

import json
import re
from schemas.generate import GeneratedTexts


def parse_gemini_response(raw_text: str) -> GeneratedTexts:
    """Gemini 응답에서 JSON을 추출하고 검증한다."""

    # 1) JSON 블록 추출 (코드 블록으로 감싸진 경우 처리)
    json_match = re.search(r'```json\s*(.*?)\s*```', raw_text, re.DOTALL)
    if json_match:
        json_str = json_match.group(1)
    else:
        # 중괄호로 시작하는 JSON 직접 추출
        json_match = re.search(r'\{.*\}', raw_text, re.DOTALL)
        if not json_match:
            raise ValueError("AI_RESPONSE_PARSE_ERROR: JSON을 찾을 수 없습니다.")
        json_str = json_match.group(0)

    # 2) JSON 파싱
    try:
        data = json.loads(json_str)
    except json.JSONDecodeError as e:
        raise ValueError(f"AI_RESPONSE_PARSE_ERROR: JSON 파싱 실패 - {e}")

    # 3) 필수 필드 검증
    required_fields = ["main_copy", "sub_copy", "body_texts", "product_descriptions", "cta_text", "hashtags"]
    missing = [f for f in required_fields if f not in data]
    if missing:
        raise ValueError(f"AI_RESPONSE_PARSE_ERROR: 필수 필드 누락 - {missing}")

    # 4) 길이 검증 (경고 수준, 자르지는 않음)
    warnings = []
    if len(data["main_copy"]) > 20:
        warnings.append(f"main_copy가 {len(data['main_copy'])}자로 권장(15자) 초과")
    if len(data["sub_copy"]) > 40:
        warnings.append(f"sub_copy가 {len(data['sub_copy'])}자로 권장(30자) 초과")

    # 5) Pydantic 모델로 변환
    return GeneratedTexts(**data), warnings


# 재시도 로직
MAX_RETRIES = 2

async def generate_text_with_retry(self, prompt: str) -> GeneratedTexts:
    """최대 2회 재시도하여 텍스트를 생성한다."""

    last_error = None
    for attempt in range(MAX_RETRIES + 1):
        try:
            message = self.client.messages.create(
                model=self.model,
                max_tokens=self.max_tokens,
                system=SYSTEM_PROMPT,
                messages=[{"role": "user", "content": prompt}]
            )
            raw = message.content[0].text
            result, warnings = parse_gemini_response(raw)

            return result

        except ValueError as e:
            last_error = e
            if attempt < MAX_RETRIES:
                continue  # 재시도
            raise

        except Exception as e:
            raise RuntimeError(f"AI_TEXT_GENERATION_FAILED: Gemini API 호출 실패 - {e}")
```

### 5.4 연동 서비스 구현

```python
# services/image_gen_service.py

import httpx
from typing import Optional
from config import settings


class ImageGenService:
    TIMEOUT = 30.0  # 이미지 생성 타임아웃 (초)
    MAX_RETRIES = 1

    def __init__(self):
        self.api_key = settings.IMAGE_GEN_API_KEY
        self.base_url = settings.IMAGE_GEN_BASE_URL

    def _headers(self) -> dict:
        return {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json",
        }

    async def generate_banner(
        self,
        brand_name: str,
        category: str,
        style_keywords: list[str],
        width: int = 860,
        height: int = 400,
    ) -> dict:
        """배너 이미지를 생성하고 URL을 반환한다."""

        prompt = self._build_prompt(brand_name, category, style_keywords)

        async with httpx.AsyncClient(timeout=self.TIMEOUT) as client:
            for attempt in range(self.MAX_RETRIES + 1):
                try:
                    response = await client.post(
                        f"{self.base_url}/generate",
                        headers=self._headers(),
                        json={
                            "prompt": prompt,
                            "width": width,
                            "height": height,
                            "style": "commercial",
                            "negative_prompt": "text, watermark, low quality, blurry",
                        },
                    )
                    response.raise_for_status()
                    data = response.json()

                    if data.get("status") != "success":
                        raise RuntimeError(f"이미지 생성 실패: {data.get('message')}")

                    return data

                except httpx.TimeoutException:
                    if attempt < self.MAX_RETRIES:
                        continue
                    raise RuntimeError("AI_IMAGE_GENERATION_FAILED: 이미지 생성 시간 초과")
                except httpx.HTTPStatusError as e:
                    raise RuntimeError(f"AI_IMAGE_GENERATION_FAILED: HTTP {e.response.status_code}")

    async def remove_background(self, image_bytes: bytes) -> dict:
        """이미지 배경을 제거하고 결과 URL을 반환한다."""

        async with httpx.AsyncClient(timeout=self.TIMEOUT) as client:
            try:
                response = await client.post(
                    f"{self.base_url}/remove-bg",
                    headers={"Authorization": f"Bearer {self.api_key}"},
                    files={"image": ("image.png", image_bytes, "image/png")},
                )
                response.raise_for_status()
                data = response.json()

                if data.get("status") != "success":
                    raise RuntimeError(f"배경 제거 실패: {data.get('message')}")

                return data

            except httpx.TimeoutException:
                raise RuntimeError("AI_IMAGE_GENERATION_FAILED: 배경 제거 시간 초과")

    def _build_prompt(self, brand_name: str, category: str, keywords: list[str]) -> str:
        """카테고리에 맞는 이미지 생성 프롬프트를 구성한다."""

        category_styles = {
            "food": "warm lighting, appetizing, clean background, premium food photography",
            "fashion": "editorial style, high fashion, clean minimal background, luxury feel",
        }
        base_style = category_styles.get(category, "professional, clean, modern")
        keyword_str = ", ".join(keywords) if keywords else ""

        return f"{brand_name} {category} promotion banner, {base_style}, {keyword_str}"
```

```python
# services/storage_service.py (이미지 저장 부분)

async def download_and_store(
    self,
    image_url: str,
    project_id: str,
    image_type: str,    # "input" | "generated" | "output"
    filename: str,
) -> str:
    """외부 URL의 이미지를 다운로드하여 Supabase Storage에 저장한다."""

    # 1. 이미지 다운로드
    async with httpx.AsyncClient(timeout=15.0) as client:
        response = await client.get(image_url)
        response.raise_for_status()
        image_bytes = response.content

    # 2. Storage 경로 구성
    storage_path = f"{project_id}/{image_type}/{filename}"

    # 3. Supabase Storage 업로드
    self.supabase.storage.from_(
        self.bucket_name
    ).upload(
        path=storage_path,
        file=image_bytes,
        file_options={"content-type": "image/png"}
    )

    return storage_path
```

### 6.1 전체 생성 흐름

```python
# services/generate_orchestrator.py

class GenerateOrchestrator:
    """AI 생성 전체 흐름을 조율한다."""

    def __init__(
        self,
        gemini_service: GeminiService,
        image_service: ImageGenService,
        storage_service: StorageService,
        db: SupabaseClient,
    ):
        self.gemini = gemini_service
        self.image = image_service
        self.storage = storage_service
        self.db = db

    async def generate(self, project_id: str, template_id: str) -> GenerateResult:
        """
        1. 프로젝트 + 템플릿 정보 조회
        2. 상태를 'generating'으로 변경
        3. 텍스트 생성 (Gemini)
        4. 이미지 생성 (Nano Banana Pro)
        5. 생성 결과를 Storage에 저장
        6. DB에 generated_content 업데이트
        7. 상태를 'editing'으로 변경
        """

        # 1. 데이터 조회
        project = await self._get_project(project_id)
        template = await self._get_template(template_id)

        # 2. 상태 변경
        await self._update_status(project_id, "generating")

        try:
            # 3. 텍스트 생성
            texts = await self._generate_texts(project, template)

            # 4. 이미지 생성 (배너)
            images = await self._generate_images(project, template)

            # 5. 결과 저장
            generated_content = {
                "texts": texts.model_dump(),
                "images": images.model_dump(),
                "generated_at": datetime.now().isoformat(),
            }

            # 6. DB 업데이트
            await self._update_project(project_id, {
                "generated_content": generated_content,
                "status": "editing",
            })

            return GenerateResult(
                project_id=project_id,
                texts=texts,
                images=images,
                generated_at=datetime.now(),
            )

        except Exception as e:
            # 실패 시 상태 변경
            await self._update_status(project_id, "failed")
            raise

    async def _generate_texts(self, project, template) -> GeneratedTexts:
        """Gemini API를 사용하여 텍스트를 생성한다."""

        prompt = build_prompt(
            template_category=template["category"],
            brand_name=project["brand_name"],
            description=project["description"],
            event_period=self._format_period(project),
            price_info=project.get("price_info"),
        )
        return await self.gemini.generate_text(prompt)

    async def _generate_images(self, project, template) -> GeneratedImages:
        """Nano Banana Pro API를 사용하여 이미지를 생성한다."""

        style_map = {
            "food": ["warm", "appetizing", "premium"],
            "fashion": ["editorial", "luxury", "minimal"],
        }
        keywords = style_map.get(template["category"], ["modern", "clean"])

        # 배너 이미지 생성
        banner_result = await self.image.generate_banner(
            brand_name=project["brand_name"],
            category=template["category"],
            style_keywords=keywords,
            width=template["width"],
            height=400,
        )

        # Storage에 저장
        banner_path = await self.storage.download_and_store(
            image_url=banner_result["image_url"],
            project_id=str(project["id"]),
            image_type="generated",
            filename="banner.png",
        )

        return GeneratedImages(
            banner=banner_path,
            products=[],  # 사용자 업로드 이미지 사용
        )
```

### 6.2 프론트엔드 생성 상태 폴링

```typescript
// hooks/useGenerate.ts

import { useState, useEffect, useCallback } from 'react';

interface GenerateState {
  status: 'idle' | 'generating' | 'completed' | 'failed';
  progress: {
    text: boolean;
    images: boolean;
    layout: boolean;
  };
  result: GenerateResult | null;
  error: string | null;
}

export function useGenerate() {
  const [state, setState] = useState<GenerateState>({
    status: 'idle',
    progress: { text: false, images: false, layout: false },
    result: null,
    error: null,
  });

  const generate = useCallback(async (projectId: string, templateId: string) => {
    setState(prev => ({ ...prev, status: 'generating', error: null }));

    try {
      // 생성 요청
      const response = await fetch(`${API_URL}/api/generate`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ project_id: projectId, template_id: templateId }),
      });

      if (!response.ok) {
        const err = await response.json();
        throw new Error(err.message || '생성에 실패했습니다.');
      }

      const result = await response.json();
      setState({
        status: 'completed',
        progress: { text: true, images: true, layout: true },
        result,
        error: null,
      });

      return result;

    } catch (err) {
      setState(prev => ({
        ...prev,
        status: 'failed',
        error: err instanceof Error ? err.message : '알 수 없는 오류가 발생했습니다.',
      }));
      throw err;
    }
  }, []);

  return { ...state, generate };
}
```

### 7.1 폰트 사전 로딩

```typescript
// lib/fonts.ts

/**
 * html-to-image 변환 전에 폰트가 로딩되어야 한다.
 * Google Fonts를 사전 로딩하는 유틸리티.
 */
export async function ensureFontsLoaded(): Promise<void> {
  const fonts = [
    { family: 'Noto Serif KR', weights: ['400', '700'] },
    { family: 'Noto Sans KR', weights: ['400', '500', '600', '700'] },
  ];

  const promises = fonts.flatMap(font =>
    font.weights.map(weight => {
      const face = new FontFace(
        font.family,
        `url(https://fonts.gstatic.com/s/${font.family.toLowerCase().replace(/ /g, '')}/v1/${weight}.woff2)`,
        { weight }
      );
      document.fonts.add(face);
      return face.load();
    })
  );

  await Promise.all(promises);
}
```

### 7.2 출력 함수

```typescript
// lib/export.ts

import { toPng, toJpeg } from 'html-to-image';
import { ensureFontsLoaded } from './fonts';

interface ExportOptions {
  format: 'png' | 'jpg' | 'pdf';
  quality: 1 | 2 | 3;         // pixelRatio
  filename: string;
}

export async function exportImage(
  element: HTMLElement,
  options: ExportOptions,
): Promise<void> {
  // 1. 폰트 로딩 확인
  await ensureFontsLoaded();

  // 2. 잠시 대기 (렌더링 안정화)
  await new Promise(r => setTimeout(r, 200));

  // 3. 이미지 변환
  const exportFn = options.format === 'jpg' ? toJpeg : toPng;
  const mimeParams = options.format === 'jpg' ? { quality: 0.92 } : {};

  const dataUrl = await exportFn(element, {
    pixelRatio: options.quality,
    cacheBust: true,          // 외부 이미지 CORS 우회
    skipAutoScale: true,
    ...mimeParams,
  });

  // 4. 다운로드
  const link = document.createElement('a');
  link.download = `${options.filename}.${options.format}`;
  link.href = dataUrl;
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
}
```

### 7.3 대용량 이미지 처리 주의사항

| 문제 | 해결 방안 |
|------|-----------|
| 860×4000 = 매우 긴 이미지 | `pixelRatio: 2`에서 1720×8000px → 메모리 확인 필요 |
| 외부 이미지 CORS | `cacheBust: true` 설정, Supabase Storage에 저장된 이미지 사용 |
| 폰트 미로딩 시 깨짐 | `ensureFontsLoaded()` 호출 후 200ms 대기 |
| 브라우저 메모리 제한 | `pixelRatio: 3`은 모바일에서 실패 가능 → 2x를 기본값으로 |
| PDF 출력 | PoC 범위에서는 제외, 추후 jsPDF 연동 가능 |

```sql
CREATE POLICY "Allow public read on project-images"
ON storage.objects FOR SELECT
USING (bucket_id = 'project-images');

CREATE POLICY "Allow public upload on project-images"
ON storage.objects FOR INSERT
WITH CHECK (bucket_id = 'project-images');

CREATE POLICY "Allow public read on templates"
ON storage.objects FOR SELECT
USING (bucket_id = 'templates');
```

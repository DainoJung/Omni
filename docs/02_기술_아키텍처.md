## 1. 시스템 개요

### 1.1 서비스 개념

사용자가 **상품 정보 + 테마**를 입력하면, **템플릿 기반**으로 **AI가 배경과 카피를 생성**하여 **편집 가능한 상세페이지**를 즉시 생성하는 시스템

> **핵심 원칙**: 템플릿 기반 렌더링 + AI 콘텐츠 생성 + 편집 가능한 결과물

### 1.2 핵심 변화 (v5.0)

| 항목 | 이전 (v4.1) | 현재 (v5.0) |
|------|-------------|-------------|
| **핵심 접근** | 동적 레이아웃 생성 | **템플릿 기반 렌더링** |
| **입력 데이터** | 브랜드, 상품, 카테고리, 토글 옵션 | **상품명, 가격, 이미지, 테마 (4개)** |
| **AI 역할** | 레이아웃 생성 + 좌표 추출 + 카피 | **배경 생성 + 카피 생성** |
| **출력 형태** | 이미지 + 텍스트 오버레이 | **편집 가능한 템플릿 결과** |
| **좌표 추출** | Gemini Vision 사용 | **불필요 (템플릿에 슬롯 정의)** |

### 5.3 AI 서비스 구현

```python
# services/template_ai_service.py

from google import genai
from google.genai import types
from typing import Dict, List
import json

class TemplateAIService:
    """템플릿 기반 AI 서비스"""

    def __init__(self, api_key: str):
        self.client = genai.Client(api_key=api_key)
        self.model_text = "gemini-2.5-pro"
        self.model_image = "gemini-3-pro-image-preview"  # Nano Banana Pro

    async def generate_campaign_copy(
        self,
        products: List[Dict],
        theme: Dict
    ) -> str:
        """캠페인 카피 생성"""

        prompt = f"""
당신은 신세계백화점의 마케팅 카피라이터입니다.

상품 정보:
{json.dumps(products, ensure_ascii=False, indent=2)}

테마: {theme['name']}
키워드: {', '.join(theme['copy_keywords'])}

요청사항:
- 테마와 어울리는 캠페인 카피 1줄 생성
- 15자 이내로 짧고 임팩트 있게
- 고급스럽고 세련된 톤
- 상품 특성 반영

응답 형식 (JSON):
{{"campaign_copy": "생성된 카피"}}
"""

        response = self.client.models.generate_content(
            model=self.model_text,
            contents=[prompt]
        )

        result = json.loads(response.text)
        return result["campaign_copy"]

    async def generate_theme_background(
        self,
        theme: Dict,
        template_size: Dict
    ) -> bytes:
        """테마 배경 이미지 생성"""

        width = template_size["width"]
        height = template_size["height"]

        # 가로세로 비율 계산
        if width == height:
            aspect_ratio = "1:1"
        elif width > height:
            aspect_ratio = "16:9"
        else:
            aspect_ratio = "3:4"

        prompt = f"""
{theme['background_prompt']}

Requirements:
- Premium department store aesthetic
- Suitable as background for product showcase
- Leave space in center for product images
- DO NOT include any text or letters
- Size: {width}x{height}px
- High quality, elegant design
"""

        response = self.client.models.generate_content(
            model=self.model_image,
            contents=[prompt],
            config=types.GenerateContentConfig(
                response_modalities=['IMAGE'],
                output_options=types.OutputOptions(
                    image_config=types.ImageConfig(
                        output_width=width,
                        aspect_ratio=aspect_ratio
                    )
                )
            )
        )

        for part in response.candidates[0].content.parts:
            if part.inline_data:
                return part.inline_data.data

        return None
```

## 6. 템플릿 렌더링 서비스

### 6.1 렌더링 로직

```python
# services/template_render_service.py

from typing import Dict, List
import json

class TemplateRenderService:
    """템플릿 렌더링 서비스"""

    def __init__(self, templates_path: str):
        self.templates_path = templates_path
        self.templates = self._load_templates()

    def _load_templates(self) -> Dict:
        """템플릿 로드"""
        templates = {}
        for template_id in ["single_product", "grid_2", "grid_3"]:
            with open(f"{self.templates_path}/{template_id}/slots.json") as f:
                templates[template_id] = json.load(f)
        return templates

    def select_template(self, product_count: int) -> str:
        """상품 개수 기반 템플릿 선택"""
        if product_count == 1:
            return "single_product"
        elif product_count == 2:
            return "grid_2"
        else:
            return "grid_3"

    def render_slots(
        self,
        template_id: str,
        slot_data: Dict
    ) -> Dict:
        """슬롯에 데이터 바인딩"""

        template = self.templates[template_id]
        rendered_slots = []

        for slot in template["slots"]:
            rendered_slot = {
                "id": slot["id"],
                "type": slot["type"],
                "position": slot["position"],
                "z_index": slot["z_index"],
                "editable": slot["editable"],
                "style": slot.get("style", {}),
                "value": slot_data.get(slot["id"]),
                "source": slot["source"]
            }
            rendered_slots.append(rendered_slot)

        return {
            "template_id": template_id,
            "size": template["size"],
            "slots": rendered_slots
        }
```

### 9.2 슬롯 편집기 컴포넌트

```tsx
// components/editor/SlotEditor.tsx

interface SlotEditorProps {
  slots: SlotValue[];
  onSlotUpdate: (slotId: string, newValue: string) => void;
  templateSize: { width: number; height: number };
}

export function SlotEditor({ slots, onSlotUpdate, templateSize }: SlotEditorProps) {
  const [selectedSlot, setSelectedSlot] = useState<string | null>(null);

  return (
    <div
      className="relative border rounded-lg overflow-hidden"
      style={{
        width: templateSize.width / 2,  // 50% 스케일
        height: templateSize.height / 2,
        aspectRatio: `${templateSize.width}/${templateSize.height}`
      }}
    >
      {/* 배경 레이어 */}
      {slots.filter(s => s.id === 'background').map(slot => (
        <img
          key={slot.id}
          src={slot.value}
          className="absolute inset-0 w-full h-full object-cover"
          style={{ zIndex: slot.position.z_index || 0 }}
        />
      ))}

      {/* 슬롯 레이어 */}
      {slots.filter(s => s.id !== 'background').map(slot => (
        <div
          key={slot.id}
          className={`absolute cursor-pointer transition-all ${
            selectedSlot === slot.id ? 'ring-2 ring-blue-500' : ''
          }`}
          style={{
            left: slot.position.x,
            top: slot.position.y,
            width: slot.position.width,
            height: slot.position.height,
            zIndex: slot.position.z_index || 1
          }}
          onClick={() => slot.editable && setSelectedSlot(slot.id)}
        >
          {slot.type === 'image' ? (
            <img src={slot.value} className="w-full h-full object-contain" />
          ) : (
            <span style={slot.style}>{slot.value}</span>
          )}

          {/* 편집 가능 표시 */}
          {slot.editable && (
            <div className="absolute top-1 right-1 bg-blue-500 text-white text-xs px-1 rounded">
              편집
            </div>
          )}
        </div>
      ))}

      {/* 선택된 슬롯 편집 패널 */}
      {selectedSlot && (
        <SlotEditPanel
          slot={slots.find(s => s.id === selectedSlot)!}
          onUpdate={(value) => onSlotUpdate(selectedSlot, value)}
          onClose={() => setSelectedSlot(null)}
        />
      )}
    </div>
  );
}
```
